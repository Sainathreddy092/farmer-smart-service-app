🚀 Basic Python Flask Starter Code
from flask import Flask, request, jsonify

app = Flask(__name__)

# Sample data
labors = [
    {"id": 1, "name": "Ravi", "work": "harvesting", "location": "village1"},
    {"id": 2, "name": "Suresh", "work": "plowing", "location": "village2"}
]

products = [
    {"id": 1, "name": "Urea Fertilizer", "price": 300},
    {"id": 2, "name": "Wheat Seeds", "price": 500}
]

machinery = [
    {"id": 1, "name": "Tractor", "rent": 1000},
    {"id": 2, "name": "Harvester", "rent": 2000}
]

@app.route('/')
def home():
    return "Smart Farmer Platform Running"

@app.route('/labors')
def get_labors():
    return jsonify(labors)

@app.route('/products')
def get_products():
    return jsonify(products)

@app.route('/machinery')
def get_machinery():
    return jsonify(machinery)

@app.route('/book', methods=['POST'])
def book_service():
    data = request.json
    return jsonify({"message": "Booking successful", "data": data})

if __name__ == '__main__':
    app.run(debug=True)
